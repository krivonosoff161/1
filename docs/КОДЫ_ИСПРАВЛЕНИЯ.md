```python
def calculate_stop_loss(self, entry_price, direction, symbol, atr):
    """
    Calculate stop loss based on ATR.
    
    CRITICAL: This function had bugs:
    1. ATR could be 0 ‚Üí Entry == Stop (DOGE error)
    2. SL too close ‚Üí 81.5% hit rate instead of 20-30%
    3. No validation of calculated SL distance
    """
    
    # Step 1: Get ATR
    if atr is None or atr < 0.0001:
        # ‚ö†Ô∏è ATR is too small or None
        logger.warning(f"ATR too small for {symbol}: {atr}. Using default.")
        atr = self._get_default_atr(symbol)
    
    logger.info(f"[SL_CALC] {symbol}: Entry={entry_price}, ATR={atr}, Direction={direction}")
    
    # Step 2: Calculate SL distance
    sl_multiplier = 2.0  # Can be adjusted per symbol
    sl_distance = atr * sl_multiplier
    
    # ‚ö†Ô∏è CRITICAL: Ensure minimum distance!
    min_distance = 0.001 * entry_price  # 0.1% of price
    if sl_distance < min_distance:
        logger.warning(f"[SL_CALC] SL too close! Adjusting from {sl_distance} to {min_distance}")
        sl_distance = min_distance
    
    # Step 3: Calculate SL price
    if direction == "long":
        stop_loss = entry_price - sl_distance
    else:  # short
        stop_loss = entry_price + sl_distance
    
    # Step 4: Validation
    if abs(entry_price - stop_loss) < 0.00001:
        logger.error(f"[SL_CALC] CRITICAL ERROR: Entry‚âàStop! Entry={entry_price}, SL={stop_loss}")
        return None  # Reject this order
    
    logger.info(f"[SL_CALC] Calculated SL: {stop_loss}, Distance: {abs(entry_price - stop_loss)}")
    
    return stop_loss


def _get_default_atr(self, symbol):
    """Get default ATR if calculation failed."""
    # Per-symbol defaults
    defaults = {
        'BTC-USDT': 150.0,
        'ETH-USDT': 15.0,
        'SOL-USDT': 1.5,
        'XRP-USDT': 0.05,
        'DOGE-USDT': 0.003
    }
    
    default = defaults.get(symbol, 1.0)
    logger.warning(f"Using default ATR for {symbol}: {default}")
    return default
```

```python
def _validate_signal_by_regime(self, signal, symbol):
    """
    Check if signal matches current market regime.
    
    Problem: System entered trades in consolidation mode before falls/rises.
    Solution: Don't enter LONG if downtrend, don't enter SHORT if uptrend.
    """
    
    # Get current market regime
    regime = self.regime_manager.get_current_regime(symbol)
    
    # Get trend indicator (e.g., SMA50)
    sma50 = self.indicators.get_sma(symbol, 50)
    current_price = signal.price
    
    logger.debug(f"[REGIME_FILTER] {symbol}: Regime={regime}, Price={current_price}, SMA50={sma50}")
    
    # Validation rule 1: Direction should match regime
    if signal.direction == "long":
        if regime == "downtrend":
            logger.info(f"[REGIME_FILTER] Reject LONG in downtrend for {symbol}")
            return False
        if current_price < sma50:  # Price below SMA50 = downtrend
            logger.info(f"[REGIME_FILTER] Reject LONG below SMA50 for {symbol}")
            return False
    
    elif signal.direction == "short":
        if regime == "uptrend":
            logger.info(f"[REGIME_FILTER] Reject SHORT in uptrend for {symbol}")
            return False
        if current_price > sma50:  # Price above SMA50 = uptrend
            logger.info(f"[REGIME_FILTER] Reject SHORT above SMA50 for {symbol}")
            return False
    
    # Validation rule 2: Volatility should be adequate
    atr = self.indicators.get_atr(symbol)
    atr_average = self.indicators.get_atr_average(symbol, period=20)
    
    if atr < atr_average * 0.5:  # Too low volatility
        logger.info(f"[REGIME_FILTER] Reject: Low volatility for {symbol} (ATR={atr})")
        return False
    
    logger.debug(f"[REGIME_FILTER] ACCEPTED {signal.direction} for {symbol}")
    return True
```

```python
def _check_order_anomalies(self, entry, stop, symbol):
    """Check if order parameters are anomalous."""
    
    # Check 1: Entry == Stop
    if abs(entry - stop) < 0.00001:
        logger.error(f"CRITICAL: Entry ‚âà Stop for {symbol}! Entry={entry}, Stop={stop}")
        return True
    
    # Check 2: SL distance too small (< 0.01% of price)
    sl_distance = abs(entry - stop)
    min_distance = entry * 0.0001  # 0.01%
    if sl_distance < min_distance:
        logger.error(f"CRITICAL: SL too close for {symbol}! Distance={sl_distance}, Min={min_distance}")
        return True
    
    # Check 3: SL distance too large (> 10% of price)
    max_distance = entry * 0.1  # 10%
    if sl_distance > max_distance:
        logger.error(f"WARNING: SL too far for {symbol}! Distance={sl_distance}, Max={max_distance}")
        return True
    
    return False  # No anomalies
```
